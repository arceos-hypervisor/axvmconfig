19/// A part of `AxVMConfig`, which represents guest VM type.
20#[derive(Default, Clone, Copy, PartialEq, Eq, Debug, serde::Serialize, serde::Deserialize)]
21pub enum VMType {
22    /// Host VM, used for boot from Linux like Jailhouse do, named "type1.5".
23    VMTHostVM = 0,
24    /// Guest RTOS, generally a simple guest OS with most of the resource passthrough.
25    #[default]
26    VMTRTOS = 1,
27    /// Guest Linux, generally a full-featured guest OS with complicated device emulation requirements.
28    VMTLinux = 2,
29}
30
31impl From<usize> for VMType {
32    fn from(value: usize) -> Self {
33        match value {
34            0 => Self::VMTHostVM,
35            1 => Self::VMTRTOS,
36            2 => Self::VMTLinux,
37            _ => {
38                warn!("Unknown VmType value: {}, default to VMTRTOS", value);
39                Self::default()
40            }
41        }
42    }
43}
44
45impl From<VMType> for usize {
46    fn from(value: VMType) -> Self {
47        value as usize
48    }
49}

81/// The type of Emulated Device.
82///
83/// Allocation scheme:
84/// - 0x00 - 0x1F: Special devices, and abstract device types that does not specify a concrete
85/// interface or implementation. The device objects created from these types depend on the target
86/// architecture and the specific implementation of the hypervisor.
87/// - 0x20 - 0x7F: Concrete emulated device types.
88///   - 0x20 - 0x2F: Interrupt controller devices.
89///   - 0x30 - 0x3F: Reserved for future use.
90/// - 0x80 - 0xDF: Reserved for future use.
91/// - 0xE0 - 0xEF: Virtio devices.
92/// - 0xF0 - 0xFF: Reserved for future use.
93#[derive(Debug, Copy, Clone, PartialEq, Eq, Serialize_repr, Deserialize_repr, Enumerable)]
94#[repr(u8)]
95pub enum EmulatedDeviceType {
96    // Special devices and abstract device types.
97    /// Dummy device type.
98    Dummy = 0x0,
99    /// Interrupt controller device, e.g. vGICv2 in aarch64, vLAPIC in x86.
100    InterruptController = 0x1,
101    /// Console (serial) device.
102    Console = 0x2,
103    /// An emulated device that provides Inter-VM Communication (IVC) channel.
104    ///
105    /// This device is used for communication between different VMs,
106    /// the corresponding memory region of this device should be marked as `Reserved` in
107    /// device tree or ACPI table.
108    IVCChannel = 0xA,
109
110    // Arch-specific interrupt controller devices.
111    // 0x20 - 0x22: GPPT (GIC Partial Passthrough) devices.
112    /// ARM GIC Partial Passthrough Redistributor device.
113    GPPTRedistributor = 0x20,
114    /// ARM GIC Partial Passthrough Distributor device.
115    GPPTDistributor = 0x21,
116    /// ARM GIC Partial Passthrough Interrupt Translation Service device.
117    GPPTITS = 0x22,
118
119    // Virtio devices.
120    /// Virtio block device.
121    VirtioBlk = 0xE1,
122    /// Virtio net device.
123    VirtioNet = 0xE2,
124    /// Virtio console device.
125    VirtioConsole = 0xE3,
126    // Following are some other emulated devices that are not currently used and removed from the enum temporarily.
127    // /// IOMMU device.
128    // IOMMU = 0x6,
129    // /// Interrupt ICC SRE device.
130    // ICCSRE = 0x7,
131    // /// Interrupt ICC SGIR device.
132    // SGIR = 0x8,
133    // /// Interrupt controller GICR device.
134    // GICR = 0x9,
135}
136
137impl Default for EmulatedDeviceType {
138    fn default() -> Self {
139        Self::Dummy
140    }
141}
142
143impl Display for EmulatedDeviceType {
144    // Implementation of the Display trait for EmulatedDeviceType.
145    fn fmt(&self, f: &mut Formatter<'_>) -> core::fmt::Result {
146        match self {
147            EmulatedDeviceType::Console => write!(f, "console"),
148            EmulatedDeviceType::InterruptController => write!(f, "interrupt controller"),
149            EmulatedDeviceType::GPPTRedistributor => {
150                write!(f, "gic partial passthrough redistributor")
151            }
152            EmulatedDeviceType::GPPTDistributor => write!(f, "gic partial passthrough distributor"),
153            EmulatedDeviceType::GPPTITS => write!(f, "gic partial passthrough its"),
154            // EmulatedDeviceType::IOMMU => write!(f, "iommu"),
155            // EmulatedDeviceType::ICCSRE => write!(f, "interrupt icc sre"),
156            // EmulatedDeviceType::SGIR => write!(f, "interrupt icc sgir"),
157            // EmulatedDeviceType::GICR => write!(f, "interrupt controller gicr"),
158            EmulatedDeviceType::IVCChannel => write!(f, "ivc channel"),
159            EmulatedDeviceType::Dummy => write!(f, "meta device"),
160            EmulatedDeviceType::VirtioBlk => write!(f, "virtio block"),
161            EmulatedDeviceType::VirtioNet => write!(f, "virtio net"),
162            EmulatedDeviceType::VirtioConsole => write!(f, "virtio console"),
163        }
164    }
165}

167/// Implementation of methods for EmulatedDeviceType.
168impl EmulatedDeviceType {
169    /// Returns true if the device is removable.
170    pub fn removable(&self) -> bool {
171        matches!(
172            *self,
173            EmulatedDeviceType::InterruptController
174                // | EmulatedDeviceType::SGIR
175                // | EmulatedDeviceType::ICCSRE
176                | EmulatedDeviceType::GPPTRedistributor
177                | EmulatedDeviceType::VirtioBlk
178                | EmulatedDeviceType::VirtioNet
179                // | EmulatedDeviceType::GICR
180                | EmulatedDeviceType::VirtioConsole
181        )
182    }
183
184    /// Converts a usize value to an EmulatedDeviceType.
185    pub fn from_usize(value: usize) -> EmulatedDeviceType {
186        match value {
187            0x0 => EmulatedDeviceType::Dummy,
188            0x1 => EmulatedDeviceType::InterruptController,
189            0x2 => EmulatedDeviceType::Console,
190            0xA => EmulatedDeviceType::IVCChannel,
191            0x20 => EmulatedDeviceType::GPPTRedistributor,
192            0x21 => EmulatedDeviceType::GPPTDistributor,
193            0x22 => EmulatedDeviceType::GPPTITS,
194            0xE1 => EmulatedDeviceType::VirtioBlk,
195            0xE2 => EmulatedDeviceType::VirtioNet,
196            0xE3 => EmulatedDeviceType::VirtioConsole,
197            // 0x6 => EmulatedDeviceType::IOMMU,
198            // 0x7 => EmulatedDeviceType::ICCSRE,
199            // 0x8 => EmulatedDeviceType::SGIR,
200            // 0x9 => EmulatedDeviceType::GICR,
201            _ => {
202                warn!("Unknown emulated device type value: {value}, default to Meta");
203                EmulatedDeviceType::Dummy
204            }
205        }
206    }
207}

241/// The configuration structure for the guest VM base info.
242#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
243pub struct VMBaseConfig {
244    /// VM ID.
245    pub id: usize,
246    /// VM name.
247    pub name: String,
248    /// VM type.
249    pub vm_type: usize,
250    // Resources.
251    /// The number of virtual CPUs.
252    pub cpu_num: usize,
253    /// The physical CPU ids.
254    /// - if `None`, vcpu's physical id will be set as vcpu id.
255    /// - if set, each vcpu will be assigned to the specified physical CPU mask.
256    ///
257    /// Some ARM platforms will provide a specified cpu hw id in the device tree, which is
258    /// read from `MPIDR_EL1` register (probably for clustering).
259    pub phys_cpu_ids: Option<Vec<usize>>,
260    /// The mask of physical CPUs who can run this VM.
261    ///
262    /// - If `None`, vcpu will be scheduled on available physical CPUs randomly.
263    /// - If set, each vcpu will be scheduled on the specified physical CPUs.
264    ///      
265    ///     For example, [0x0101, 0x0010] means:
266    ///          - vCpu0 can be scheduled at pCpu0 and pCpu2;
267    ///          - vCpu1 will only be scheduled at pCpu1;
268    ///      It will phrase an error if the number of vCpus is not equal to the length of `phys_cpu_sets` array.
269    pub phys_cpu_sets: Option<Vec<usize>>,
270}

303/// Specifies how the VM should handle interrupts and interrupt controllers.
304#[derive(Debug, Clone, Copy, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
305pub enum VMInterruptMode {
306    /// The VM will not handle interrupts, and the guest OS should not use interrupts.
307    #[serde(rename = "no_irq", alias = "no", alias = "none")]
308    NoIrq,
309    /// The VM will use the emulated interrupt controller to handle interrupts.
310    #[serde(rename = "emu", alias = "emulated")]
311    Emulated,
312    /// The VM will use the passthrough interrupt controller (including GPPT) to handle interrupts.
313    #[serde(rename = "passthrough", alias = "pt")]
314    Passthrough,
315}
316
317impl Default for VMInterruptMode {
318    fn default() -> Self {
319        Self::NoIrq
320    }
321}

335/// The configuration structure for the guest VM serialized from a toml file provided by user,
336/// and then converted to `AxVMConfig` for the VM creation.
337#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
338pub struct AxVMCrateConfig {
339    /// The base configuration for the VM.
340    pub base: VMBaseConfig,
341    /// The kernel configuration for the VM.
342    pub kernel: VMKernelConfig,
343    /// The devices configuration for the VM.
344    pub devices: VMDevicesConfig,
345}
346
347impl AxVMCrateConfig {
348    /// Deserialize the toml string to `AxVMCrateConfig`.
349    pub fn from_toml(raw_cfg_str: &str) -> AxResult<Self> {
350        let config: AxVMCrateConfig = toml::from_str(raw_cfg_str).map_err(|err| {
351            warn!("Config TOML parse error {:?}", err.message());
352            axerrno::ax_err_type!(InvalidInput, alloc::format!("Error details {err:?}"))
353        })?;
354        Ok(config)
355    }
356}
