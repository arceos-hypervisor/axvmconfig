15/// A part of `AxVMConfig`, which represents guest VM type.
16#[derive(Default, Clone, Copy, PartialEq, Eq, Debug, serde::Serialize, serde::Deserialize)]
17pub enum VMType {
18    /// Host VM, used for boot from Linux like Jailhouse do, named "type1.5".
19    VMTHostVM = 0,
20    /// Guest RTOS, generally a simple guest OS with most of the resource passthrough.
21    #[default]
22    VMTRTOS = 1,
23    /// Guest Linux, generally a full-featured guest OS with complicated device emulation requirements.
24    VMTLinux = 2,
25}
26
27impl From<usize> for VMType {
28    fn from(value: usize) -> Self {
29        match value {
30            0 => Self::VMTHostVM,
31            1 => Self::VMTRTOS,
32            2 => Self::VMTLinux,
33            _ => {
34                warn!("Unknown VmType value: {}, default to VMTRTOS", value);
35                Self::default()
36            }
37        }
38    }
39}
40
41impl From<VMType> for usize {
42    fn from(value: VMType) -> Self {
43        value as usize
44    }
45}

47/// The type of memory mapping.
48#[derive(Debug, Clone, PartialEq, Eq, serde_repr::Serialize_repr, serde_repr::Deserialize_repr)]
49#[repr(u8)]
50pub enum VmMemMappingType {
51    /// The memory region is allocated by the VM monitor.
52    MapAlloc = 0,
53    /// The memory region is identical to the host physical memory region.
54    MapIentical = 1,
55}
56
57/// The default value of `VmMemMappingType` is `MapAlloc`.
58impl Default for VmMemMappingType {
59    fn default() -> Self {
60        Self::MapAlloc
61    }
62}

109/// The configuration structure for the guest VM base info.
110#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
111pub struct VMBaseConfig {
112    /// VM ID.
113    pub id: usize,
114    /// VM name.
115    pub name: String,
116    /// VM type.
117    pub vm_type: usize,
118    // Resources.
119    /// The number of virtual CPUs.
120    pub cpu_num: usize,
121    /// The physical CPU ids.
122    /// - if `None`, vcpu's physical id will be set as vcpu id.
123    /// - if set, each vcpu will be assigned to the specified physical CPU mask.
124    ///
125    /// Some ARM platforms will provide a specified cpu hw id in the device tree, which is
126    /// read from `MPIDR_EL1` register (probably for clustering).
127    pub phys_cpu_ids: Option<Vec<usize>>,
128    /// The mask of physical CPUs who can run this VM.
129    ///
130    /// - If `None`, vcpu will be scheduled on available physical CPUs randomly.
131    /// - If set, each vcpu will be scheduled on the specified physical CPUs.
132    ///      
133    ///     For example, [0x0101, 0x0010] means:
134    ///          - vCpu0 can be scheduled at pCpu0 and pCpu2;
135    ///          - vCpu1 will only be scheduled at pCpu1;
136    ///      It will phrase an error if the number of vCpus is not equal to the length of `phys_cpu_sets` array.
137    pub phys_cpu_sets: Option<Vec<usize>>,
138}

180/// The configuration structure for the guest VM serialized from a toml file provided by user,
181/// and then converted to `AxVMConfig` for the VM creation.
182#[derive(Debug, Default, Clone, serde::Serialize, serde::Deserialize)]
183pub struct AxVMCrateConfig {
184    /// The base configuration for the VM.
185    pub base: VMBaseConfig,
186    /// The kernel configuration for the VM.
187    pub kernel: VMKernelConfig,
188    /// The devices configuration for the VM.
189    pub devices: VMDevicesConfig,
190}
191
192impl AxVMCrateConfig {
193    /// Deserialize the toml string to `AxVMCrateConfig`.
194    pub fn from_toml(raw_cfg_str: &str) -> AxResult<Self> {
195        let config: AxVMCrateConfig = toml::from_str(raw_cfg_str).map_err(|err| {
196            warn!("Config TOML parse error {:?}", err.message());
197            axerrno::ax_err_type!(InvalidInput, alloc::format!("Error details {err:?}"))
198        })?;
199        Ok(config)
200    }
